SELECT parameter_name, CASE WHEN data_type='USER-DEFINED' THEN udt_name ELSE trim(both '\"' from data_type) END AS data_type, parameter_mode, ordinal_position
FROM information_schema.parameters
WHERE specific_schema='{0}' AND specific_name='{1}'
UNION
SELECT 'return_type' AS parameter_name, CASE WHEN data_type='USER-DEFINED' THEN udt_name ELSE trim(both '\"' from data_type) END AS data_type, 'RETURN' AS parameter_mode, -1 AS ordinal_position
FROM information_schema.routines
WHERE specific_schema='{0}' AND specific_name='{1}'
ORDER BY ordinal_position ASC
